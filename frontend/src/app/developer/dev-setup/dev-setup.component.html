Getting Started with kt-component

For Angular

1) Add Custom Elements Schema in your project including
CUSTOM_ELEMENTS_SCHEMA
in
src/app/app.module.ts
import {{ '{' }} BrowserModule {{ '}' }} from '@angular/platform-browser';
import {{ '{' }} NgModule, CUSTOM_ELEMENTS_SCHEMA {{ '}' }} from '@angular/core';
import {{ '{' }} AppComponent {{ '}' }} from './app.component';

@NgModule({{ '{' }}
  declarations: [
    AppComponent
  ],
  imports: [
    BrowserModule
  ],
  providers: [],
  bootstrap: [AppComponent],
  schemas: [
    CUSTOM_ELEMENTS_SCHEMA
  ]
})

export class AppModule {{ '{' }} {{ '}' }}
2) Install
kt-component
in your project.
npm install
2) Import
kt-component
in
src/app/app.module.ts
.
import 'kt-component'

For React

1) Install
kt-component
in your project.
npm install
2) Import
kt-component
in
src/App.js
import 'kt-component'

For Vue.js

1) Add
vue.config.js
in the root folder of the project and do this for linked modules.
module.exports = {{ '{' }}
chainWebpack: config => config.resolve.symlinks(false)
{{ '}' }}
2) Install
kt-component
in your project.
npm install
3) Import
kt-component
in
src/App.vue
import 'kt-component'
